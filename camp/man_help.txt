/*WHALE OS COMMANDS*/

COMMAND NAME     COMMAND DEFINTION

	pwd          --print your working directory

	ls           --Lists files and directories @PWD

	cat          --copies file to specified destination

	edit         --pulls up editor window to edit code

	cd           --change the directory you are in

	build        --builds a binary of your program

	flash        --loads the specified binary to connected 

	man          --opens up the manual for the specified page

	test         --runs test routine on WISDOM hardware

	send         --sends a user input string to the specified radio
